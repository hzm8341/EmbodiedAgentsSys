import sys
sys.path.insert(0, '/media/hzm/data_disk/ros-agents')

import yaml
import re
from pathlib import Path
from dataclasses import dataclass, field
from typing import List, Dict, Any

# 复制以下代码到 Python REPL 或 test_parser.py

@dataclass
class OpenClawSkillConfig:
    name: str
    description: str
    metadata: Dict[str, Any] = field(default_factory=dict)
    rules: List[str] = field(default_factory=list)
    references: Dict[str, str] = field(default_factory=dict)

class OpenClawSkillParser:
    @staticmethod
    def parse(skill_dir: str) -> OpenClawSkillConfig:
        skill_path = Path(skill_dir)
        content = (skill_path / "SKILL.md").read_text()
        
        # 解析 frontmatter
        pattern = r'^---\n(.*?)\n---'
        match = re.match(pattern, content, re.DOTALL)
        frontmatter = yaml.safe_load(match.group(1))
        
        config = OpenClawSkillConfig(
            name=frontmatter.get("name", ""),
            description=frontmatter.get("description", ""),
            metadata=frontmatter.get("metadata", {})
        )
        
        # 解析 rules
        pattern = r'##\s*生成代码规则\s*\n(.*?)(?:\n##|\Z)'
        match = re.search(pattern, content, re.DOTALL)
        if match:
            config.rules = [l.strip() for l in match.group(1).split('\n') if l.strip().startswith(('.', '-'))]
        
        # 读取 references
        ref_dir = skill_path / "references"
        if ref_dir.exists():
            for f in ref_dir.glob("*.md"):
                config.references[f.stem] = f.read_text()
        
        return config

# 执行测试
config = OpenClawSkillParser.parse("/media/hzm/data_disk/openclaw_robot_control/skill/agx_arm_codegen")

# 验证

print(f"Name: {config.name}")
print(f"Description: {config.description[:30]}...")
print(f"Metadata: {config.metadata}")
print(f"Rules count: {len(config.rules)}")
print(f"References: {list(config.references.keys())}")

def generate_template(description: str, robot_type: str = "nero") -> str:
    return f'''#!/usr/bin/env python3
"""
Generated by agx-arm-codegen Skill
Description: {description}
Robot: {robot_type}
"""

import time
from pyAgxArm import create_agx_arm_config, AgxArmFactory


def wait_motion_done(robot, timeout: float = 3.0, poll_interval: float = 0.01) -> bool:
    start_t = time.monotonic()
    while True:
        status = robot.get_arm_status()
        if status is not None and getattr(status.msg, "motion_status", None) == 0:
            return True
        if time.monotonic() - start_t > timeout:
            return False
        time.sleep(poll_interval)


def main():
    robot_cfg = create_agx_arm_config(
        robot="{robot_type}",
        comm="can",
        channel="can0",
        interface="socketcan",
    )
    robot = AgxArmFactory.create_arm(robot_cfg)
    robot.connect()
    
    time.sleep(1)
    robot.set_normal_mode()
    time.sleep(1)
    
    while not robot.enable():
        time.sleep(0.01)
    
    robot.set_speed_percent(80)
    
    # === 用户请求: {description} ===
    robot.set_motion_mode(robot.MOTION_MODE.J)
    robot.move_j([0.0] * 7)
    wait_motion_done(robot)


if __name__ == "__main__":
    main()
'''

# 复制以下代码

class CodeValidator:
    RISKY_PATTERNS = [
        ("move_js", "move_js 是快速响应模式，请谨慎使用"),
        ("move_mit", "MIT 模式是高风险操作"),
    ]
    
    @staticmethod
    def validate(code: str):
        result = {"is_valid": True, "errors": [], "warnings": []}
        
        # 检查导入
        if "pyAgxArm" not in code:
            result["errors"].append("缺少 pyAgxArm 导入")
            result["is_valid"] = False
        
        # 检查 main 函数
        if "def main" not in code:
            result["errors"].append("缺少 main 函数")
            result["is_valid"] = False
        
        # 检查 enable
        if "robot.enable" not in code:
            result["warnings"].append("缺少 enable 调用")
        
        # 检查风险模式
        for pattern, msg in CodeValidator.RISKY_PATTERNS:
            if pattern in code:
                result["warnings"].append(f"风险: {msg}")
        
        return result


# === 测试用例 ===

# 用例 1: 有效代码
valid_code = """import time
from pyAgxArm import create_agx_arm_config, AgxArmFactory

def main():
    robot_cfg = create_agx_arm_config(robot="nero")
    robot = AgxArmFactory.create_arm(robot_cfg)
    robot.connect()
    while not robot.enable():
        time.sleep(0.01)
    robot.set_normal_mode()
"""

print("=== 用例 1: 有效代码 ===")
r1 = CodeValidator.validate(valid_code)
print(f"结果: {r1}")
assert r1["is_valid"] == True
print("✓ 通过\n")

    print("hello")

print("=== 用例 2: 无效代码 ===")
r2 = CodeValidator.validate(invalid_code)

print("✓ 正确拒绝\n")

# 用例 3: 风险代码

def main():
    robot.move_js([0,0,0,0,0,0,0])

import asyncio
import re
import yaml
from pathlib import Path
from dataclasses import dataclass, field
from typing import Dict, Any, List, Optional
from enum import Enum

# === 复制必要的类 (从前面复制) ===
class SkillStatus:
    IDLE = "idle"
    RUNNING = "running"
    SUCCESS = "success"
    FAILED = "failed"

@dataclass
class SkillResult:
    status: str
    output: Any = None
    error: Optional[str] = None
    metadata: Dict[str, Any] = field(default_factory=dict)

# Parser 和 Generator 复制过来...

class AgxArmCodeGenSkill:
    def __init__(self, skill_dir=None, llm_client=None, validate_code=True):
        self.config = OpenClawSkillParser.parse(skill_dir or "/media/hzm/data_disk/openclaw_robot_control/skill/agx_arm_codegen")
        self.llm_client = llm_client
        self.validate_code = validate_code
    
    async def execute(self, user_description: str, robot_type: str = "nero"):
        try:
            # 生成代码
            if self.llm_client:
                response = await self.llm_client.generate("prompt")
                code = self._extract_code(response)
            else:
                code = CodeGenerator.generate_template(user_description, robot_type)
            
            # 验证
            validation = None
            if self.validate_code:
                validation = CodeValidator.validate(code)
            
            return SkillResult(
                status=SkillStatus.SUCCESS,
                output={
                    "generated_code": code,
                    "description": user_description,
                    "validation": validation
                },
                metadata={"skill_name": self.config.name}
            )
        except Exception as e:
            return SkillResult(status=SkillStatus.FAILED, error=str(e))
    
    def _extract_code(self, response: str) -> str:
        pattern = r'```python\n(.*?)```'
        match = re.search(pattern, response, re.DOTALL)
        return match.group(1).strip() if match else response.strip()

# 复制 OpenClawSkillManager 类...

class OpenClawSkillManager:
    def __init__(self, skills_base_dir=None):
        self.skills_base_dir = Path(skills_base_dir or "/media/hzm/data_disk/openclaw_robot_control/skill")
        self.loaded_skills = {}
    
    def discover_skills(self):
        if not self.skills_base_dir.exists():
            return []
        return [item.name for item in self.skills_base_dir.iterdir() 
                if item.is_dir() and (item / "SKILL.md").exists()]
    
    def load_skill(self, skill_name):
        config = OpenClawSkillParser.parse(str(self.skills_base_dir / skill_name))
        self.loaded_skills[skill_name] = config
        return config

# === 执行测试 ===

manager = OpenClawSkillManager()

print("=== 测试 1: 发现 Skills ===")
skills = manager.discover_skills()
print(f"发现: {skills}")
assert "agx_arm_codegen" in skills

print("\n=== 测试 2: 加载 Skill ===")
config = manager.load_skill("agx_arm_codegen")
print(f"名称: {config.name}")
print(f"规则数: {len(config.rules)}")

print("\n=== 测试 3: 获取实例 ===")
